6 /* Scheduler - Wouter Stokman - 2025418@umail.leidenuniv.nl
7  * Contains implementation of a simple class that enable scheduling of schedule_units.
8  * Scheduler is called every x seconds, where x is the time in seconds, given when calling scheduler_start_main_ecore_loop(x)
9  * This ensures that the scheduler_update function is called every x seconds with the current time
10  *
11  * When a schedule_unit is due to be executed (unit.timestamp < curtime during a call to scheduler_update), the callback function that is specified in the schedule_unit is called, along with a pointer to the to-be-executed schedule_unit
12  *
13  * After this, the schedule unit is deleted from the scheduler, after which it can be manually re-added in the callback function to repeatedly reschedule and re-execute a schedule unit (e.g. execute every 10 seconds).
14  *
15  *
16  *
17  *
18  * Example:
19  *
20  * void simpleCallback(schedule_unit * unit){
21  * printf(unit.nparam); //prints the number of parameters of the unit that is being executed
22  * unit.timestamp+=1000; //execute the unit in a second
23  * scheduler_insert(unit); //insert a new copy of the unit into the scheduler so it will be executed again, the executed unit will be deleted after this
24  *
25  * }
26  *
27  *
28  * schedule_unit unit;
29  * unit.id = scheduler_get_new_unit_id(); //call this function to get a new ID that has not been used before (enabling safe deletion by id)
30  * unit.unit_execute_function = simpleCallback; //set the callback function
31  * unit.nparam = .....; //number of parameters
32  * unit.path = ..... //set path here
33  * unit.param = ..... //set param here
34  * unit.timestamp = curtime + 1000; //Curtime=milliseconds since epoch, this will ensure unit will be executed one seconds from now
35  *
36  * scheduler_initialize(); //initialize the scheduler
37  * scheduler_start_main_encore_loop(1.5); //update scheduler every 1.5 seconds
38  * scheduler_insert(&unit); //insert a copy of the created unit in the scheduler
39  * //This is all that needs to be done, new schedule units can then be added/deleted
40  *
41  */
